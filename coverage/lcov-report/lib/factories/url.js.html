module.exports.flexibleUrl = function() {
  return function(prefix) {
    return function(suffix) {
      if (suffix) {
        return prefix + '/' + suffix;
      } else {
        return prefix;
      }
    };
  };
};
 
module.exports.urlApi = function(flexibleUrl) {
  return flexibleUrl('/api');
};
